export interface Ingredient {
  name: string;
  amount: string; // "1/2", "3", "to taste"
  unit: string;   // "cup", "tbsp", "grams", "pieces"
}

export interface Recipe {
  id: string;            // Hash of source URL
  title: string;
  sourceUrl: string;
  extractedAt: string;   // ISO timestamp
  servings: string;      // "2-4" or "4"
  prepTime: string;      // "15 mins"
  cookTime: string;      // "30 mins"
  ingredients: Ingredient[];
  instructions: string[]; // Ordered step array
}

export type LoadingStage = 'idle' | 'fetching_stream' | 'transcribing_audio' | 'parsing_ai' | 'error';

export interface APIRequestPayload {
  url: string;
  useMock: boolean;
}

export interface APIResponsePayload {
  success: boolean;
  recipe?: Recipe;
  error?: string;
  modelUsed?: string;
}

Shared Utilities (Copy These)
Regex Engine
Save to: src/utils/regex.ts
/**
 * Extracts clean Instagram Reel URL from messy mobile share text.
 * Input: "Look at this! https://www.instagram.com/reel/C8XyZ/?igsh=MTZ1&other=param"
 * Output: "https://www.instagram.com/reel/C8XyZ/"
 */
export function extractInstagramUrl(sharedText: string): string | null {
  const urlRegex = /(https?:\/\/(?:www\.)?instagram\.com\/(?:p|reel)\/[a-zA-Z0-9_\-]+)/i;
  const match = sharedText.match(urlRegex);
  if (!match) return null;
  return `${match[1].replace(/\/$/, '')}/`;
}

Dev 1: Backend Pipeline
Target File: src/app/api/extract/route.ts
Job: Ingest Instagram URL → stream video via Cobalt → transcribe with Groq → extract recipe with LLM cascade → return clean JSON.
Codex Prompt (Copy-Paste This):
Create src/app/api/extract/route.ts as a Next.js 14 Route Handler.

Exports:
export const maxDuration = 300; // 5-minute timeout for Vercel
export const dynamic = 'force-dynamic';

POST Handler Logic:
1. Accept { url: string, useMock: boolean }
2. If useMock === true: return { success: true, recipe: mockRecipe }
3. If useMock === false:
   a) POST to https://api.cobalt.tools/api/json with body:
      { "url": "INPUT_URL", "downloadMode": "audio", "audioFormat": "mp3" }
      Grab response.url (direct stream link)
   b) Fetch audio stream to ArrayBuffer, convert to Blob
   c) POST to Groq Whisper (https://api.groq.com/openai/v1/audio/transcriptions)
      Headers: Authorization: Bearer ${GROQ_API_KEY}
      Body: FormData with file + model: "whisper-large-v3"
      Extract response.text (transcript)
   d) POST to OpenRouter (https://openrouter.ai/api/v1/chat/completions)
      Try in order: llama-3.3-70b -> qwen-2.5-72b -> gemma-2-9b -> GPT-4o fallback
      System prompt: "Extract recipe from transcript. Return ONLY valid JSON (no markdown) matching this schema: {title, servings, prepTime, cookTime, ingredients: [{name, amount, unit}], instructions: [string[]]}"
      User prompt: "Transcription: [INSERT TEXT]"
      Catch 429 errors, retry next model in chain
   e) Parse JSON response strictly. If parsing fails, retry next model.
   f) Return { success: true, recipe: PARSED_JSON, modelUsed: "model_name" }

Error handling: Wrap everything in try-catch. Return { success: false, error: "message" } on failure.
Environment Variables:
GROQ_API_KEY=your_groq_key
OPENROUTER_API_KEY=your_openrouter_key
OPENAI_API_KEY=fallback_only
Integration Point: Dev 2 calls this with fetch('/api/extract', { method: 'POST', body: JSON.stringify({ url, useMock: true/false }) }).

2. Service Worker
Save to: public/sw.js
const CACHE_NAME = 'r2r-v1-core';
const ASSETS = ['/', '/manifest.json'];

self.addEventListener('install', (e) => {
  e.waitUntil(caches.open(CACHE_NAME).then(c => c.addAll(ASSETS)));
  self.skipWaiting();
});

self.addEventListener('activate', (e) => {
  e.waitUntil(clients.claim());
});

self.addEventListener('fetch', (e) => {
  e.respondWith(
    caches.match(e.request).then(res => res || fetch(e.request))
  );
});

4. Share Target Router
Save to: src/app/shared/page.tsx
'use client';

import { useSearchParams, useRouter } from 'next/navigation';
import { useEffect } from 'react';
import { extractInstagramUrl } from '@/utils/regex';

export default function SharedPage() {
  const searchParams = useSearchParams();
  const router = useRouter();

  useEffect(() => {
    const sharedText = searchParams.get('text');

    if (!sharedText) {
      router.replace('/');
      return;
    }

    const cleanedUrl = extractInstagramUrl(sharedText);

    if (!cleanedUrl) {
      router.replace('/?error=invalid_link');
      return;
    }

    // Route back to dashboard with URL pre-filled
    router.replace(`/?shareUrl=${encodeURIComponent(cleanedUrl)}`);
  }, [searchParams, router]);

  return <div>Processing share...</div>;
}
